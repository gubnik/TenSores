409 std::size_t calculateIndex(const std::array<std::size_t, Order>& dims) const
410 {
411 std::shared_lock<std::shared_mutex> lock(mutex());
412 std::size_t index = 0;
413 std::size_t multiplier = 1;
414 for (std::size_t i = 0; i < Order; ++i) {
415 if (dims[i] >= m_DimensionsData[i]) {
416 throw std::out_of_range("Index out of bounds");
417 }
418 index += dims[i] * multiplier;
419 multiplier *= m_DimensionsData[i];
420 }
421 return index;
422 }
