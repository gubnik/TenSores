407 std::size_t calculateIndex(const std::array<std::size_t, Order>& dims) const
408 {
409 std::shared_lock<std::shared_mutex> lock(mutex());
410 std::size_t index = 0;
411 std::size_t multiplier = 1;
412 for (std::size_t i = 0; i < Order; ++i) {
413 if (dims[i] >= m_DimensionsData[i]) {
414 throw std::out_of_range("Index out of bounds");
415 }
416 index += dims[i] * multiplier;
417 multiplier *= m_DimensionsData[i];
418 }
419 return index;
420 }
