356 std::size_t calculateIndex(const std::array<std::size_t, Order>& dims) const
357 {
358 std::lock_guard<std::shared_mutex> lock(mutex());
359 std::size_t index = 0;
360 std::size_t multiplier = 1;
361 for (std::size_t i = 0; i < Order; ++i) {
362 if (dims[i] >= m_DimensionsData[i]) {
363 throw std::out_of_range("Index out of bounds");
364 }
365 index += dims[i] * multiplier;
366 multiplier *= m_DimensionsData[i];
367 }
368 return index;
369 }
