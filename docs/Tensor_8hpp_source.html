412 std::size_t calculateIndex(const std::array<std::size_t, Order>& dims) const
413 {
414 std::lock_guard<std::shared_mutex> lock(mutex());
415 std::size_t index = 0;
416 std::size_t multiplier = 1;
417 for (std::size_t i = 0; i < Order; ++i) {
418 if (dims[i] >= m_DimensionsData[i]) {
419 throw std::out_of_range("Index out of bounds");
420 }
421 index += dims[i] * multiplier;
422 multiplier *= m_DimensionsData[i];
423 }
424 return index;
425 }
